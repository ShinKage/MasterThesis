\newcommand{\MemoryLayout}[1]{
  \coordinate (start) at (0, 0);
  \foreach \lab/\dr [remember=\lab as \p (initially start)] in {#1} {
    \node[draw, rectangle, minimum height=1.25cm, minimum width=1cm, right of=\p] (\lab) [] {$\lab$};
  }
  \node[draw, rectangle, minimum height=1.25cm, minimum width=1cm, right of=\last{#1}] (ell) [] {$\dots$};
  \node[draw, rectangle, color=white, very thick, minimum height=1.25cm, minimum width=1pt, right=-1pt of ell] () [] {};
  \draw[](\head{#1}.north west)++(0.7pt,0)--++(0,24pt)node[above]{$0$};
}
